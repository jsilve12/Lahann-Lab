
Create Table news(
	pk integer auto_increment not null,
	title varchar(256),
	dat date,
	author varchar(64),
	contents varchar(4096),
	primary key (pk)
) engine = innoDB;
Create Table images(
	pk integer auto_increment not null,
	art integer,
	name varchar(256),
	primary key(pk),
	constraint fk_article
		foreign key (art)
		references news(pk)
) engine = innoDB;
